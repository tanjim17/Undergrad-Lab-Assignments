Opcode:

G -> 0000 -> 0 -> or
H -> 0001 -> 1 -> ori
C -> 0010 -> 2 -> sub
I -> 0011 -> 3 -> sll
A -> 0100 -> 4 -> add
F -> 0101 -> 5 -> andi
E -> 0110 -> 6 -> and
B -> 0111 -> 7 -> addi
J -> 1000 -> 8 -> srl
K -> 1001 -> 9 -> nor
D -> 1010 -> 10 -> subi
P -> 1011 -> 11 -> jump(j)
L -> 1100 -> 12 -> sw
N -> 1101 -> 13 -> beq
O -> 1110 -> 14 -> bneq
M -> 1111 -> 15 -> lw


registers:

0 -> $zero
1 -> $t0
2 -> $t1
3 -> $t2
4 -> $t3
5 -> $t4
6 -> $sp
